Naming

Variables

Start first word lowercase and after that all words start with uppercase letter:

    var foo = 'bar',
        fooBar = true;

Constants

Use uppercase:

    var MYCONST = 'i\'m  your father'

Functions

Start first word lowercase and after that all words start with uppercase letter:

      function foo() {}
      function fooBar() {}
      function veryLongFunctionName() {}

Use words like 'is', 'set', 'get':

      function isReady() {}
      function getProperty() {}
      function setValue() {}

Arrays

Use plural forms:

      var objects = [{}, {}, {}];

Objects

All words start with uppercase letter:

      var Object = {},
          UsefullObject = {'PI' : 3.14};

      function ConstructorFunction(args) {}

Tips & Tricks

new Type()

Do not use:

      var o = new Object(),
          a = new Array(),
          n = new Number(1),
          s = new String('1'),
          b = new Boolean(true);

Be simple, it's faster

      var o = {},
          a = [],
          n = 1,
          s = '1',
          b = true;

if

Do not use:

     if (foo === true) {
      // do smth
     }

Instead

      if (foo) {
        //do smth
      }

Ternary

Do not use:

    var stuff;
    if (foo === 'bar') {
      stuff = 10
    } else {
      stuff = 0
    }

Instead

     var stuff = foo === 'bar' ? 10 : 0;

But

Do not use:

    var isEqual = a === b ? true : false;

Instead

    var isEqual = a === b;

for

    var messages = document.querySelectorAll('.message');
    for (var i = 0; i < messages.length; i++) {
      // do smth with messages[i]
    }

Much better

    var messages = document.querySelectorAll('.message');
    for (var i = 0, n = messages.length; i < n; i++) {
      // do smth with messages[i]
    }

even better

    var messages = document.querySelectorAll('.message'),
        n = messages.length,
        i = 0;
    for (; i < n; i++) {
      // do smth with messages[i]
    }

Or even

    var messages = document.querySelectorAll('.message'),
        i = messages.length;
    while (i--) {
      // do smth with messages[i];
    }

for in

    for (prop in someObject) {
      // do smth with someObject[prop]
    }

Use only if you're sure!

Think twice before using for in

Early returns

function returnLate(foo) {
  var ret;
  if (foo) {
    ret = 'foo';
  } else {
    ret = 'quux';
  }
  return ret; // Bad
}

function returnEarly(foo) {
  if (foo) {
    return 'foo';
  }
  return 'quux'; // Good
}

function returnImmediate() {
  return foo ? 'foo' : 'quux'; // Or even this way
}

HTML strings

      var div = '<div class="wrapper">';
        div += '<p>some content</p>';
        div += '</div>';
        //1 reflow

Still better than

      $('<div>').addClass('wrapper').append('<p>').text('some content')
      // 4 reflows

jQuery variables

      // Start with $ 
      var $buttons = $('button')
      // Always cache your variables
      var $wrapper = $('#wrapper');
      
      $('.message').text(someText);
      $('.message').on('click', function() {}); // Very bad

      // Use chaining
      $('.message').text(someText).on('click', function() {});
      // Or increase readability
      $('.message')
      .text(someText)
      .on('click', function() {});

jQuery selectors

      $('li[data-selected="true"] a') // Looks good, but weeeery slow
      $('li.selected a')              // Much Better
      $('#elem')                      // Fastest
      $('a.button:animated'); // Does not use querySelectorAll()
      $('a.button').filter(':animated');  // Uses it
