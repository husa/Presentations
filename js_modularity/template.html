AMD

Asynchronous Module Definition

A Format For Writing Modular JavaScript In The Browser
  • born out of XHR+eval
  • module and dependencies can be asynchronously loaded
  • all magic is inside a lib
  • require
  • define

define

define(
    module_id,  /* optional */
    [dependencies],  /* optional */
    definition function /*function for instantiating the module or object*/
);

define example

define('myModule',
    ['foo', 'bar'],
    // module definition function
    // dependencies (foo and bar) are mapped to function parameters
    function ( foo, bar ) {
        // return a value that defines the module export
        // (i.e the functionality we want to expose for consumption)

        // create your module here
        var myModule = {
            doStuff:function(){
                console.log('Yay! Stuff');
            }
        }

        return myModule;
});

require

require is typically used to load code in a top-level JavaScript file

require(['foo', 'bar'], function ( foo, bar ) {
        // rest of your code here
        foo.doSomeStuff();
        bar.doAnotherStuff();
});

dynamic require

define(function ( require ) {
    var isReady = false, foobar;

    // note the inline require within our module definition
    require(['foo', 'bar'], function (foo, bar) {
        isReady = true;
        foobar = foo() + bar();
    });

    // we can still return a module
    return {
        isReady: isReady,
        foobar: foobar
    };
});

Why AMD?

  • approach to define flexible modules
  • better than <script> tag
  • module definitions are encapsulated
  • doesn't have issues with cross-domain, local or debugging
  • lazy load

CommonJS
  • no function wrapper around module(no "define")
  • exports - object that is available for others
  • require - import another module

require() and exports

// package/lib is a dependency we require
var lib = require('package/lib');

// some behaviour for our module
function foo(){
    lib.log('hello world!');
}

// export (expose) foo to other modules
exports.foo = foo;

the same in AMD

define(['package/lib'], function(lib){

    // some behaviour for our module
    function foo(){
        lib.log('hello world!');
    }

    // export (expose) foo for other modules
    return {
        foobar: foo
    };
});

basic usage

// define more behaviour we would like to expose
function foobar(){
        this.foo = function(){
                console.log('Hello foo');
        }

        this.bar = function(){
                console.log('Hello bar');
        }
}
// expose foobar to other modules
exports.foobar = foobar;


// an application consuming 'foobar'
var foobar = require('./foobar').foobar,
    test   = new foobar();

test.bar(); // 'Hello bar'

multiple exports

exports.app = function(){
    console.log('hello from app');
}

exports.foo = function(){
    console.log('hello from foo');
}

ES Harmony

Declare a Module

module Math {
    export function sum(x, y) {
        return x + y;
    }
    export var pi = 3.141593;
}

Use Module

// specific functionality
import Math.{sum, pi};

console.log(sum(2,2));

// all functionality
import Math.*;

console.log(sum(2,2));

//rename imported functionality
import Math.{sum : calcSum, pi : PI};
